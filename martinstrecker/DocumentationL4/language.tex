\begin{syntaxdiagram}

\node[junction,right=of start] (p1) {};
\node[nonterminal,right=of p1] (term) {term};
\node[terminal,below=of term] (plus)  {+};
\node[junction,right=of term] (p2) {};
\node[junction,right=of p2] (end) {};
\node[terminal,below=of plus] (longer)  {longer};

\graph [use existing nodes] {
start -- p1 -> term -- p2 -> end;
};

\graph [use existing nodes] {
p2 ->[vh path] plus;
plus ->[hv path] p1;
p2 ->[vh path] longer;
longer ->[hv path] p1;
};
\end{syntaxdiagram}
